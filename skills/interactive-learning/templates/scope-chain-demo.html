📝 对应代码


// 全局作用域
const globalVar = "我是全局变量";

function outer() {
    // outer 作用域
    const x = 10;

    function inner() {
        // inner 作用域 - 当前执行位置
        const y = 20;
        console.log(x);  // 作用域链查找：inner → outer → 找到x=10
    }

    return inner;
}

const fn = outer();
fn();  // 执行inner函数
